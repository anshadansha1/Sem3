#include<stdio.h>
int a=10;
void demo()
{
	static int b=1;
	int c=1;
	printf("\na=%d, b=%d, c=%d",a,b,c);
	a++;
	b++;
	c++;
}
int main()
{
	register int d=10;
	demo();
	printf("\na=%d, d=%d",a,d);
	//printf("\nb=%d, c=%d",b,c); Wont work since local variable
	//printf("\nAdress of c= %d",&c); Wont work since register variables have no adress
	demo();
	demo();
	return 0;
}
